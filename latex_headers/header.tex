\newcommand{\thesistitlepage}{
    \normalspace
    \thispagestyle{empty}
    \begin{center}
        \textbf{\LARGE \thesistitle} \\[1cm]
        \textbf{\LARGE \thesisauthor} \\[8cm]
        \ \\
        \ \\
        \ \\
        \ \\[3cm]
        \textbf{\Large COLUMBIA UNIVERSITY} \\[4mm]
        \thesisyear
    \end{center}
    \clearpage
}

    \end{center}

}
